public class AuthServiceProxy implements AuthService {
    
    private final Function<String, Optional<ValidateResponse>> resilientValidator;

    public AuthServiceProxy(...) {
        // Create the reusable "Resilient Pipeline" once during initialization
        this.resilientValidator = Decorators.ofFunction((String token) -> callAuthService(token))
            .withThreadPoolBulkhead(threadPoolBulkhead) // Isolated threads
            .withTimeLimiter(timeLimiter, scheduler)   // Timeout + Interruption
            .withCircuitBreaker(circuitBreaker)        // Failure tracking
            .withRetry(retry)                          // Auto-retry
            .decorate();
    }

    @Override
    public Optional<ValidateResponse> validateToken(String token) {
        try {
            // Simply call the pre-built pipeline
            return resilientValidator.apply(token);
        } catch (Exception e) {
            return handleExceptions(e);
        }
    }
}
